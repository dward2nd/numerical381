function GaussSeidel(A, b, x, iter = 10)
    for j in 1:iter
        for i in 1:length(b)
            x[i] = 0
            x[i] = (b[i] - sum(x .* A[i, :])) / A[i, i]
        end
    end
    
    return x
end

function isDDM(A)
    row_n = size(A, 1)
    gt_count = 0
    
    for i in 1:row_n
        s = 0
        for j in 1:row_n
            if i == j
                continue
            end
            s += abs(A[i, j])
        end
        
        if abs(A[i, i]) > s
            gt_count += 1
        elseif abs(A[i, i]) < s
            return false
        end
    end
    
    return gt_count > 0
end
